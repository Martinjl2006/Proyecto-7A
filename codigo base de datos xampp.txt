CREATE DATABASE LegendAR;
USE LegendAR;

CREATE TABLE Provincias (
    id_provincia INT PRIMARY KEY AUTO_INCREMENT,
    Coordenada_long DECIMAL(9,6) NOT NULL,
    Coordenada_lat DECIMAL(9,6) NOT NULL,
    Nombre VARCHAR(100) NOT NULL
);

CREATE TABLE Ciudad (
    id_ciudad INT PRIMARY KEY AUTO_INCREMENT,
    Nombre VARCHAR(150) NOT NULL,
    Coordenada_long DECIMAL(9,6) NOT NULL,
    Coordenada_lat DECIMAL(9,6) NOT NULL,
    id_provincia INT NOT NULL,
    FOREIGN KEY (id_provincia) REFERENCES Provincias(id_provincia)
);

CREATE TABLE Usuarios (
    id_usuario INT PRIMARY KEY AUTO_INCREMENT,
    Nombre VARCHAR(100) NOT NULL,
    mail VARCHAR(255) NOT NULL,
    apellido VARCHAR(100) NOT NULL,
    Username VARCHAR(100) NOT NULL,
    contrasena VARCHAR(100) NOT NULL,
    id_provincia INT NOT NULL,
    id_ciudad INT NOT NULL,
    FOREIGN KEY (id_provincia) REFERENCES Provincias(id_provincia),
    FOREIGN KEY (id_ciudad) REFERENCES Ciudad(id_ciudad)
);

CREATE TABLE MitoLeyenda (
    id_mitooleyenda INT PRIMARY KEY AUTO_INCREMENT,
    Titulo VARCHAR(255) NOT NULL,
    Descripcion TEXT NOT NULL,
    Fecha DATE NOT NULL,
    foto VARCHAR(255), -- ahora solo guarda el nombre de archivo
    id_ciudad INT NOT NULL,
    id_provincia INT NOT NULL,
    id_usuario INT NOT NULL,
    FOREIGN KEY (id_ciudad) REFERENCES Ciudad(id_ciudad),
    FOREIGN KEY (id_provincia) REFERENCES Provincias(id_provincia),
    FOREIGN KEY (id_usuario) REFERENCES Usuarios(id_usuario)
);
